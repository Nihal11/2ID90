\section{The algorithm}

\begin{function}[H]
	\caption{alphabeta(node, depth, $\alpha$, $\beta$)}
		\If{depth = 0 or node is a terminal node}{
    	    return the heuristic value of node
    	}
    	\eIf{node.player = white}{
    		\For{each child of node}{
    			$\alpha$ =  max($\alpha$, alphabeta(child, depth - 1, $\alpha$, $\beta$))\\
            	\If{$beta \leq alpha$}{
					break \tcc*{Beta cut-off}
            	}
			}
        	\Return{$\alpha$}
    	}{ % else
    		\For{each child of node}{
    			$\beta$ = min($\beta$, alphabeta(child, depth - 1, $\alpha$, $\beta$))\\
				\If{$\alpha \leq \beta$}{
					break \tcc*{Alpha cut-off}
				}
			}
			\Return{$\beta$}
		}
\end{function}
